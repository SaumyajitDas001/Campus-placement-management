7. Student Skills Queries
-------------------------

SELECT
    s.roll_no,
    s.full_name,
    sk.name AS skill,
    ss.proficiency
FROM student_skills ss
JOIN students s ON s.student_id = ss.student_id
JOIN skills sk ON sk.skill_id = ss.skill_id
ORDER BY s.roll_no, sk.name
LIMIT 100;

SELECT
    s.roll_no,
    s.full_name,
    string_agg(sk.name || ' (' || ss.proficiency || ')', ', ' ORDER BY sk.name) AS skills
FROM student_skills ss
JOIN students s ON s.student_id = ss.student_id
JOIN skills sk ON sk.skill_id = ss.skill_id
GROUP BY s.roll_no, s.full_name
ORDER BY s.roll_no
LIMIT 50;

SELECT
    sk.name AS skill,
    COUNT(*) AS student_count
FROM student_skills ss
JOIN skills sk ON sk.skill_id = ss.skill_id
GROUP BY sk.name
ORDER BY student_count DESC;


8. Semester And Subject Marks Queries
-------------------------------------

SELECT
    s.roll_no,
    s.full_name,
    sub.semester_no,
    sub.subject_code,
    sub.subject_name,
    sub.credits,
    m.internal_marks,
    m.external_marks,
    m.total_marks,
    m.grade,
    m.exam_status
FROM student_semester_marks m
JOIN students s ON s.student_id = m.student_id
JOIN subjects sub ON sub.subject_id = m.subject_id
WHERE s.roll_no = 'CSE0001'
ORDER BY sub.semester_no, sub.subject_code;

SELECT
    s.roll_no,
    s.full_name,
    sub.semester_no,
    ROUND(AVG(m.total_marks), 2) AS average_marks
FROM student_semester_marks m
JOIN students s ON s.student_id = m.student_id
JOIN subjects sub ON sub.subject_id = m.subject_id
GROUP BY s.roll_no, s.full_name, sub.semester_no
ORDER BY s.roll_no, sub.semester_no
LIMIT 80;

SELECT
    s.roll_no,
    s.full_name,
    sub.semester_no,
    ROUND(
        SUM(
            CASE m.grade
                WHEN 'O' THEN 10
                WHEN 'A+' THEN 9
                WHEN 'A' THEN 8
                WHEN 'B+' THEN 7
                WHEN 'B' THEN 6
                WHEN 'C' THEN 5
                WHEN 'P' THEN 4
                ELSE 0
            END * sub.credits
        )::numeric / NULLIF(SUM(sub.credits), 0),
        2
    ) AS sgpa
FROM student_semester_marks m
JOIN students s ON s.student_id = m.student_id
JOIN subjects sub ON sub.subject_id = m.subject_id
WHERE s.roll_no = 'CSE0001'
GROUP BY s.roll_no, s.full_name, sub.semester_no
ORDER BY sub.semester_no;

SELECT
    s.roll_no,
    s.full_name,
    COUNT(*) AS backlog_subjects
FROM student_semester_marks m
JOIN students s ON s.student_id = m.student_id
WHERE m.exam_status = 'BACKLOG'
GROUP BY s.roll_no, s.full_name
ORDER BY backlog_subjects DESC, s.roll_no
LIMIT 50;

SELECT
    sub.semester_no,
    sub.subject_code,
    sub.subject_name,
    ROUND(AVG(m.total_marks), 2) AS class_average
FROM student_semester_marks m
JOIN subjects sub ON sub.subject_id = m.subject_id
GROUP BY sub.semester_no, sub.subject_code, sub.subject_name
ORDER BY sub.semester_no, sub.subject_code;


9. Company Queries
------------------

SELECT *
FROM companies
ORDER BY package_lpa DESC;

SELECT
    c.name AS company,
    c.package_lpa,
    c.openings,
    c.min_cgpa,
    c.max_backlogs,
    string_agg(DISTINCT d.code, ', ' ORDER BY d.code) AS allowed_departments

JOIN departments d ON d.department_id = cd.department_id
GROUP BY c.company_id, c.name, c.package_lpa, c.openings, c.min_cgpa, c.max_backlogs

FROM company_required_skills crs
JOIN companies c ON c.company_id = crs.company_id
JOIN skills sk ON sk.skill_id = crs.skill_id
ORDER BY c.name, crs.weight DESC;

FROM v_company_dashboard

SELECT

SELECT *
FROM interview_slots
WHERE booked_count < capacity
ORDER BY lower(slot_range);

14. Bipartite Matching / Matching Candidate Queries
---------------------------------------------------

-- Meaning:
-- Each row is a possible edge between one eligible student and one available interview slot.
-- This is useful for maximum bipartite matching algorithms.

SELECT *
FROM v_matching_edges
ORDER BY match_score DESC
LIMIT 100;

SELECT
    e.student_id,
    s.roll_no,
    s.full_name,
    c.name AS company,
    e.slot_id,
    lower(sl.slot_range) AS starts_at,
    upper(sl.slot_range) AS ends_at,
    e.match_score
FROM v_matching_edges e
JOIN students s ON s.student_id = e.student_id
JOIN companies c ON c.company_id = e.company_id
JOIN interview_slots sl ON sl.slot_id = e.slot_id
ORDER BY e.match_score DESC, c.name, s.roll_no
LIMIT 100;

SELECT
    company_id,
    COUNT(*) AS possible_edges
FROM v_matching_edges
GROUP BY company_id
ORDER BY possible_edges DESC;

16. Transactional Interview Booking Queries
-------------------------------------------

-- Book one interview safely using the PL/pgSQL function:

SELECT book_interview(1, 1, 1);

-- Example using selected IDs:

SELECT book_interview(
    (SELECT student_id FROM students WHERE roll_no = 'CSE0001'),
    (SELECT company_id FROM companies WHERE name = 'TCS'),
    (SELECT slot_id FROM interview_slots WHERE company_id = (SELECT company_id FROM companies WHERE name = 'TCS') LIMIT 1)
);

-- Manual transaction style:

BEGIN;

SELECT *
FROM interview_slots
WHERE slot_id = 1
FOR UPDATE;

SELECT *
FROM students
WHERE student_id = 1
FOR UPDATE;

COMMIT;

-- Rollback if needed:

ROLLBACK;


17. Interval Scheduling / Overlap Detection Queries
---------------------------------------------------

-- Check whether a student has overlapping interviews:

SELECT
    s.roll_no,
    s.full_name,
    sl1.slot_id AS slot_1,
    sl2.slot_id AS slot_2,
    sl1.slot_range AS range_1,
    sl2.slot_range AS range_2
FROM interview_schedules sch1
JOIN interview_schedules sch2
    ON sch1.student_id = sch2.student_id
   AND sch1.schedule_id < sch2.schedule_id
JOIN students s ON s.student_id = sch1.student_id
JOIN interview_slots sl1 ON sl1.slot_id = sch1.slot_id
JOIN interview_slots sl2 ON sl2.slot_id = sch2.slot_id
WHERE sl1.slot_range && sl2.slot_range;

-- Find available slots for a company:

SELECT
    sl.slot_id,
    c.name AS company,
    lower(sl.slot_range) AS starts_at,
    upper(sl.slot_range) AS ends_at,
    sl.capacity,
    sl.booked_count
FROM interview_slots sl
JOIN companies c ON c.company_id = sl.company_id
WHERE c.name = 'Microsoft'
  AND sl.booked_count < sl.capacity
ORDER BY starts_at;

20. Analytics Queries
---------------------

SELECT
    d.code AS department,
    COUNT(*) AS total_students,
    ROUND(AVG(s.cgpa), 2) AS average_cgpa
FROM students s
JOIN departments d ON d.department_id = s.department_id
GROUP BY d.code
ORDER BY total_students DESC;

SELECT
    d.code AS department,
    COUNT(*) FILTER (WHERE s.backlogs = 0) AS no_backlog_students,
    COUNT(*) FILTER (WHERE s.backlogs > 0) AS students_with_backlogs
FROM students s
JOIN departments d ON d.department_id = s.department_id
GROUP BY d.code
ORDER BY d.code;

SELECT
    placement_status,
    COUNT(*) AS total
FROM students
GROUP BY placement_status;

SELECT
    c.name AS company,
    c.package_lpa,
    COUNT(DISTINCT a.student_id) AS applicants,
    COUNT(DISTINCT sch.student_id) AS scheduled_students
FROM companies c
LEFT JOIN applications a ON a.company_id = c.company_id
LEFT JOIN interview_schedules sch ON sch.company_id = c.company_id
GROUP BY c.name, c.package_lpa
ORDER BY c.package_lpa DESC;

SELECT
    company_name,
    ROUND(AVG(match_score), 2) AS average_match_score,
    MAX(match_score) AS best_match_score,
    MIN(match_score) AS lowest_match_score
FROM v_eligible_students
GROUP BY company_name
ORDER BY average_match_score DESC;
